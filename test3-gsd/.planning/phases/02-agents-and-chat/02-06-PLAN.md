    // src/server/streaming.ts
    // SDK event → AI SDK 6 UIMessageChunk adapter per AI-SPEC §3.2 / RESEARCH §3.2.
    // Five event-mapping rules + three custom data-* chunks (tool-trace, wiki-citation, claim-id).
    // Hooks are non-blocking per RESEARCH landmine #15 — ToolTraceSink is an EventEmitter.

    import { EventEmitter } from 'node:events';
    import { logger } from '@/lib/log.js';

    // UIMessageChunk shapes per AI SDK 6.
    // The exact field names may vary slightly across versions; verify against installed `ai` package types
    // (see node_modules/ai/dist/*.d.ts). The shape below matches the AI-SPEC §3.2 contract.
    export type UIMessageChunk =
      | { type: 'text-delta'; text: string }
      | { type: 'data-tool-trace'; value: { phase: 'start' | 'result'; tool: string; args?: unknown; summary?: string; agentId?: string } }
      | { type: 'data-wiki-citation'; value: { topicSlug: string; excerpt: string; vaultRelPath: string } }
      | { type: 'data-claim-id'; value: { claimId: string; sourceTool: string } }
      | { type: 'finish' }
      | { type: 'error'; error: string };

    // Constructor helpers — used by the chat route AND any in-stream injection point
    export function createToolTraceChunk(phase: 'start' | 'result', tool: string, args?: unknown, summary?: string, agentId?: string): UIMessageChunk {
      return { type: 'data-tool-trace', value: { phase, tool, args, summary, agentId } };
    }

    export function createWikiCitationChunk(topicSlug: string, excerpt: string, vaultRelPath: string): UIMessageChunk {
      return { type: 'data-wiki-citation', value: { topicSlug, excerpt, vaultRelPath } };
    }

    export function createClaimIdChunk(claimId: string, sourceTool: string): UIMessageChunk {
      return { type: 'data-claim-id', value: { claimId, sourceTool } };
    }

    /**
     * Maps a Claude Agent SDK event to a UIMessageChunk.
     * Returns null for events that do not surface to the UI (e.g. internal SDK lifecycle events).
     * Caller must filter nulls before sending over SSE.
     */
    export function adaptToUIMessageChunk(sdkEvent: unknown): UIMessageChunk | null {
      const ev = sdkEvent as Record<string, any>;
      if (!ev || typeof ev !== 'object') return null;

      // Rule 1: text-delta passthrough
      if (ev.type === 'text-delta' && typeof ev.text === 'string') {
        return { type: 'text-delta', text: ev.text };
      }
      // Some SDK versions emit 'content_block_delta' with delta.text — handle both
      if (ev.type === 'content_block_delta' && ev.delta?.type === 'text_delta' && typeof ev.delta.text === 'string') {
        return { type: 'text-delta', text: ev.delta.text };
      }

      // Rule 2: tool-call-start → data-tool-trace (phase=start)
      if (ev.type === 'tool-call-start' || ev.type === 'tool_use') {
        return createToolTraceChunk('start', ev.tool ?? ev.name ?? 'unknown', ev.args ?? ev.input, undefined, ev.agentId);
      }

      // Rule 3: tool-call-result → data-tool-trace (phase=result)
      if (ev.type === 'tool-call-result' || ev.type === 'tool_result') {
        const summary = typeof ev.result === 'string'
          ? ev.result.slice(0, 80)
          : (ev.summary ?? JSON.stringify(ev.result ?? ev.content)?.slice(0, 80));
        return createToolTraceChunk('result', ev.tool ?? ev.name ?? 'unknown', undefined, summary, ev.agentId);
      }

      // Rule 4: message-end / done
      if (ev.type === 'message-end' || ev.type === 'message_stop' || ev.type === 'done' || ev.type === 'finish') {
        return { type: 'finish' };
      }

      // Rule 5: errors
      if (ev.type === 'error' || ev.error) {
        const errMsg = typeof ev.error === 'string' ? ev.error : (ev.error?.message ?? JSON.stringify(ev.error));
        return { type: 'error', error: errMsg ?? 'unknown SDK error' };
      }

      // Unrecognized event — log debug, return null (filtered out before SSE send)
      logger.debug({ ev }, 'streaming.adaptToUIMessageChunk: unrecognized event shape');
      return null;
    }

    /**
     * Non-blocking trace sink for SDK hooks per RESEARCH landmine #15.
     * Hooks emit-and-return only; this sink buffers events for the chat route to forward over SSE.
     */
    export class ToolTraceSink extends EventEmitter {
      emit(event: string | symbol, ...args: unknown[]): boolean {
        // EventEmitter.emit is synchronous; consumers attach via .on('event', listener)
        return super.emit(event, ...args);
      }
    }

    export const globalToolTraceSink = new ToolTraceSink();
    ```

    // src/server/routes/chat.ts
    // POST /chat — Hono SSE handler bridging runCoordinatorTurn to AssistantChatTransport.
    // RESEARCH §INFRA-04 + §3.2; AI-SPEC §3 lines 218-277.
    // Tool-event matchers use FULL MCP-prefixed tool IDs (no substring matching).

    import { Hono } from 'hono';
    import { streamSSE } from 'hono/streaming';
    import { runCoordinatorTurn } from '@/agents/coordinator.js';
    import { adaptToUIMessageChunk, createClaimIdChunk, globalToolTraceSink } from '../streaming.js';
    import { applyOutputGuard } from '@/agents/coordinator-output-guard.js';
    import { logger } from '@/lib/log.js';

    // Tool ID constants — match exactly. Do NOT substring-match.
    // The `mcp__<server>__<tool>` form is set by createSdkMcpServer + tool() in src/agents/tools/.
    const TOOL_ONEBRAIN_WRITE_CLAIM = 'mcp__onebrain__onebrain_write_claim';
    const SUB_AGENT_RESEARCH = 'research';      // SDK ctx.agentId / event.agentId field

    export const chatRoute = new Hono();

    chatRoute.post('/chat', async (c) => {
      const body = await c.req.json().catch(() => ({}));
      const message: string = body?.message ?? body?.prompt ?? body?.userMessage ?? '';
      if (!message || typeof message !== 'string') {
        return c.json({ error: 'missing required body field: message (string)' }, 400);
      }
      logger.info({ messageLength: message.length }, 'POST /chat');

      return streamSSE(c, async (stream) => {
        let accumulatedReply = '';
        let lastSubAgentSummary: string | undefined;
        const claimIds: string[] = [];

        try {
          for await (const ev of runCoordinatorTurn(message)) {
            const chunk = adaptToUIMessageChunk(ev);
            if (!chunk) continue;

            // Track for output-guard
            if (chunk.type === 'text-delta') {
              accumulatedReply += chunk.text;
            } else if (chunk.type === 'data-tool-trace' && chunk.value.phase === 'result') {
              const tool = chunk.value.tool;
              const agentId = chunk.value.agentId;

              // Capture sub-agent summary if this result came from the research sub-agent.
              // Match on the EXACT agentId field — never on tool substring.
              if (agentId === SUB_AGENT_RESEARCH && typeof chunk.value.summary === 'string') {
                lastSubAgentSummary = chunk.value.summary;
              }

              // Capture written claim IDs from onebrain_write_claim results.
              // The tool ID match MUST be exact (full MCP prefix); the summary parser extracts
              // the ULID after the literal `claim:` prefix that the wrapper emits.
              if (tool === TOOL_ONEBRAIN_WRITE_CLAIM && typeof chunk.value.summary === 'string' && chunk.value.summary.startsWith('claim:')) {
                const id = chunk.value.summary.slice('claim:'.length).trim();
                if (id) {
                  claimIds.push(id);
                  // Forward as a structured data-claim-id chunk so the UI can render the inline citation.
                  await stream.writeSSE({ data: JSON.stringify(createClaimIdChunk(id, tool)) });
                }
              }
            }

            await stream.writeSSE({ data: JSON.stringify(chunk) });
          }

          // Apply output guard over accumulated reply (D-06 last-line-of-defense)
          if (lastSubAgentSummary) {
            const guard = applyOutputGuard(accumulatedReply, lastSubAgentSummary, claimIds);
            if (guard.violation) {
              // Emit a system message + the rewritten reply
              await stream.writeSSE({ data: JSON.stringify({ type: 'data-tool-trace', value: { phase: 'result', tool: 'guardrail.prose_smuggling', summary: 'reply rewritten by output guard' } }) });
              await stream.writeSSE({ data: JSON.stringify({ type: 'text-delta', text: '\n\n' + guard.reply }) });
            }
          }

          await stream.writeSSE({ data: JSON.stringify({ type: 'finish' }) });
        } catch (err) {
          logger.error({ err }, 'chat SSE error');
          await stream.writeSSE({ data: JSON.stringify({ type: 'error', error: String((err as Error).message) }) });
        }
      });
    });
    ```

    async function readSse(res: Response, maxChunks = 20): Promise<string> {
      const reader = res.body!.getReader();
      const decoder = new TextDecoder();
      let raw = '';
      let count = 0;
      while (count < maxChunks) {
        const { value, done } = await reader.read();
        if (done) break;
